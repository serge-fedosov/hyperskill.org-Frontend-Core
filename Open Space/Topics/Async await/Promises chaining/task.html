Promises chaining

In the code below, we have a simple example of Promise chaining.

function animalSpecie(animal) {
  return new Promise((resolve, reject) => {
    if (animal === 'Bird') {
      resolve(animal);
    } else {
      reject(animal);
    }
  });
}
animalSpecie('Bird')
  .then((result) => console.log(`The animal is a ${result}.`))
  .then(() => console.log('This animal can fly.'))
  .then(() => console.log('This animal has feathers.'))
  .catch((result) => {
    console.log(`The animal is a ${result}.`);
    console.log("This animal can't fly.");
});
